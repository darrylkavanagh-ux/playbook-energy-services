import rateLimit from 'express-rate-limit';

import xss from 'xss-clean';
import mongoSanitize from 'express-mongo-sanitize';
import hpp from 'hpp';

export const setupSecurity = (app: Express) => {
  // 1. Security Headers (Helmet)
  app.use(helmet({
    contentSecurityPolicy: {
      directives: {
        defaultSrc: ["'self'"],
        styleSrc: ["'self'", "'unsafe-inline'", 'https://fonts.googleapis.com'],
        scriptSrc: ["'self'"],
        imgSrc: ["'self'", 'data:', 'https:'],
        connectSrc: ["'self'", 'https://api.bailii.org', 'https://eur-lex.europa.eu'],
        fontSrc: ["'self'", 'https://fonts.gstatic.com'],
        frameSrc: ["'none'"],
      },
    },
    hsts: {
      maxAge: 31536000,
      includeSubDomains: true,
      preload: true
    },
    frameguard: { action: 'deny' },
    noSniff: true,
    xssFilter: true
  }));
  
  // 2. Rate Limiting
  const apiLimiter = rateLimit({
    windowMs: 15 * 60 * 1000, // 15 minutes
    max: 100, // limit each IP to 100 requests per windowMs
    message: 'Too many requests from this IP, please try again later.',
    standardHeaders: true,
    legacyHeaders: false,
    handler: (req, res) => {
      logger.warn('Rate limit exceeded', {
        ip: req.ip,
        path: req.path,
        userAgent: req.get('user-agent')
      });
      res.status(429).json({
        error: 'Too many requests',
        retryAfter: res.getHeader('Retry-After')
      });
    }
  });
  
  // Stricter rate limiting for sensitive endpoints
  const forensicLimiter = rateLimit({
    windowMs: 15 * 60 * 1000,
    max: 50, // More restrictive for forensic operations
    message: 'Forensic operation rate limit exceeded'
  });
  
  const authLimiter = rateLimit({
    windowMs: 15 * 60 * 1000,
    max: 5, // Very restrictive for auth attempts
    message: 'Too many authentication attempts'
  });
  
  app.use('/api/', apiLimiter);
  app.use('/api/forensic/', forensicLimiter);
  app.use('/api/auth/', authLimiter);
  
  // 3. CORS Configuration
  const corsOptions = {
    origin: (origin: string, callback: Function) => {
      const whitelist = process.env.CORS_ORIGIN?.split(',') || [];
      if (!origin || whitelist.includes(origin) || process.env.NODE_ENV === 'development') {
        callback(null, true);
      } else {
        callback(new Error('Not allowed by CORS'));
      }
    },
    credentials: true,
    optionsSuccessStatus: 200,
    methods: ['GET', 'POST', 'PUT', 'DELETE', 'PATCH'],
    allowedHeaders: ['Content-Type', 'Authorization', 'X-Requested-With']
  };
  
  app.use(cors(corsOptions));
  
  // 4. XSS Protection
  app.use(xss());
  
  // 5. NoSQL Injection Protection
  app.use(mongoSanitize());
  
  // 6. HTTP Parameter Pollution Protection
  app.use(hpp({
    whitelist: ['sort', 'filter', 'page', 'limit']
  }));
  
  // 7. Additional Security Headers
  app.use((req, res, next) => {
    res.setHeader('X-Content-Type-Options', 'nosniff');
    res.setHeader('X-Frame-Options', 'DENY');
    res.setHeader('X-XSS-Protection', '1; mode=block');
    res.setHeader('Referrer-Policy', 'strict-origin-when-cross-origin');
    res.setHeader('Permissions-Policy', 'geolocation=(), microphone=(), camera=()');
    next();
  });
};

// server/src/middleware/auth.ts
import jwt from 'jsonwebtoken';
import { Request, Response, NextFunction } from 'express';

interface JWTPayload {
  userId: string;
  role: string;
  clearanceLevel: 'Confidential' | 'Secret' | 'Top Secret';
}

export const authenticateJWT = (req: Request, res: Response, next: NextFunction) => {
  const authHeader = req.headers.authorization;
  
  if (!authHeader) {
    return res.status(401).json({ error: 'No authentication token provided' });
  }
  
  const token = authHeader.split(' ')[1]; // Bearer <token>
  
  try {
    const payload = jwt.verify(token, process.env.JWT_SECRET!) as JWTPayload;
    req.user = payload;
    next();
  } catch (error) {
    logger.warn('JWT verification failed', { error, ip: req.ip });
    return res.status(403).json({ error: 'Invalid or expired token' });
  }
};

export const requireClearance = (level: string) => {
  return (req: Request, res: Response, next: NextFunction) => {
    const clearanceLevels = ['Confidential', 'Secret', 'Top Secret'];
    const requiredIndex = clearanceLevels.indexOf(level);
    const userIndex = clearanceLevels.indexOf(req.user.clearanceLevel);
    
    if (userIndex >= requiredIndex) {
      next();
    } else {
      logger.warn('Insufficient clearance', {
        userId: req.user.userId,
        required: level,
        actual: req.user.clearanceLevel
      });
      res.status(403).json({ error: 'Insufficient clearance level' });
    }
  };
};

// Request logging middleware
export const requestLogger = (req: Request, res: Response, next: NextFunction) => {
  const start = Date.now();
  
  res.on('finish', () => {
    const duration = Date.now() - start;
    
    logger.info('HTTP Request', {
      method: req.method,
      url: req.url,
      status: res.statusCode,
      duration: `${duration}ms`,
      ip: req.ip,
      userAgent: req.get('user-agent'),
      userId: req.user?.userId,

    // Log slow requests

        url: req.url,
        duration: `${duration}ms`
      });
    }
  });
  
  next();
};

// Forensic operation logger
export const logForensicOperation = (
  operation: string,
  userId: string,
  details: any
) => {
  logger.info('Forensic Operation', {
    context: 'forensic',
    operation,
    userId,
    timestamp: new Date().toISOString(),
    details,
    // Immutable audit trail
    auditHash: generateAuditHash(operation, userId, details)
  });
};

function generateAuditHash(operation: string, userId: string, details: any): string {
  const crypto = require('crypto');
  const data = JSON.stringify({ operation, userId, details, timestamp: Date.now() });
  return crypto.createHash('sha256').update(data).digest('hex');
}

// server/src/services/backup.ts
import { exec } from 'child_process';
import { promisify } from 'util';
import { S3Client, PutObjectCommand } from '@aws-sdk/client-s3';
import fs from 'fs/promises';

const execAsync = promisify(exec);

export class BackupService {
  private s3Client: S3Client;
  
  constructor() {
    this.s3Client = new S3Client({
      region: process.env.AWS_REGION || 'eu-west-1',
      credentials: {
        accessKeyId: process.env.AWS_ACCESS_KEY_ID!,
        secretAccessKey: process.env.AWS_SECRET_ACCESS_KEY!
      }
    });
  }
  
  /**
   * Full database backup
   */
  async backupDatabase(): Promise<{ success: boolean; filename: string; size: number }> {
    const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
    const filename = `orb-ai-backup-${timestamp}.sql`;
    const localPath = `/tmp/${filename}`;
    
    try {
      logger.info('Starting database backup', {
        context: 'forensic',
        timestamp
      });
      
      // Create backup using pg_dump
      const command = `pg_dump ${process.env.DATABASE_URL} > ${localPath}`;
      await execAsync(command);
      
      // Compress backup
      await execAsync(`gzip ${localPath}`);
      const compressedPath = `${localPath}.gz`;
      
      // Get file size
      const stats = await fs.stat(compressedPath);
      
      // Upload to S3
      const fileContent = await fs.readFile(compressedPath);
      await this.s3Client.send(new PutObjectCommand({
        Bucket: process.env.BACKUP_BUCKET || 'orb-ai-backups',
        Key: `database/${filename}.gz`,
        Body: fileContent,
        ServerSideEncryption: 'AES256',
        StorageClass: 'STANDARD_IA', // Infrequent Access
        Metadata: {
          platform: 'orb-ai',
          type: 'database-backup',
          timestamp: new Date().toISOString()
        }
      }));
      
      logger.info('Database backup completed', {
        context: 'forensic',
        filename,
        size: stats.size,
        location: 's3'
      });
      
      // Cleanup local file
      await fs.unlink(compressedPath);
      
      return {
        success: true,
        filename,
        size: stats.size
      };
    } catch (error) {
      logger.error('Database backup failed', {
        context: 'forensic',
        error
      });
      throw error;
    }
  }
  
  /**
   * Backup forensic evidence files
   */
  async backupEvidenceFiles(): Promise<void> {
    const evidenceDir = process.env.EVIDENCE_STORAGE_PATH || './evidence';
    const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
    const archiveName = `evidence-${timestamp}.tar.gz`;
    
    // Create tar.gz archive
    await execAsync(`tar -czf /tmp/${archiveName} ${evidenceDir}`);
    
    // Upload to S3
    const fileContent = await fs.readFile(`/tmp/${archiveName}`);
    await this.s3Client.send(new PutObjectCommand({
      Bucket: process.env.BACKUP_BUCKET || 'orb-ai-backups',
      Key: `evidence/${archiveName}`,
      Body: fileContent,
      ServerSideEncryption: 'AES256',
      StorageClass: 'GLACIER', // Long-term archival
    }));
    
    // Cleanup
    await fs.unlink(`/tmp/${archiveName}`);
  }
  
  /**
   * Schedule automated backups
   */
  scheduleBackups(): void {
    // Database backup every 6 hours
    setInterval(() => {
      this.backupDatabase().catch(error => {
        logger.error('Scheduled backup failed', { error });
      });
    }, 6 * 60 * 60 * 1000);
    
    // Evidence files backup daily
    setInterval(() => {
      this.backupEvidenceFiles().catch(error => {
        logger.error('Evidence backup failed', { error });
      });
    }, 24 * 60 * 60 * 1000);
    
    logger.info('Backup schedule activated', {
      database: '6 hours',
      evidence: '24 hours'
    });
  }
}
